signed __int64 __cdecl SHF(unsigned __int8 *a1, int a2)
{
  v4 = 3134132LL;
  for ( i = 0; i < a2; ++i )
  {
    LODWORD(v4) = a1[i] ^ (unsigned int)v4;
    v4 *= 6745203LL;
  }
  return v4;
}
